37 static __inline__ int same_dpl(unsigned short sel1, unsigned short sel2) {
38  /* Comparar los bits 0 y 1 de los selectores. Si son iguales,
39  * los selectores tienen el mismo DPL.
40  * Recuerde que el formato de un selector es:
41  * bits 0 - 1: DPL
42  * bit 2: Indicador de tabla (0 = gdt, 1 = ldt)
43  * bits 3-15 : indice del selector.
44  * Asi, al realizar la operacion 'and' con el valor 0x03, se enmascaran
45  * solo los bits correspondientes al DPL. Luego los dos valores se
46  * comparan para determinar si son iguales. */
47  return ((sel1 & 0x03) == (sel2 & 0x03));
48 }
